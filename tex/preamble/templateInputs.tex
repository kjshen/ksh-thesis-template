\newcommand{\varLanguage}{english,ngerman}
% Declare all languages that will be needed in the document as comma separated list.
% The *last* language is considered the main language:
%   For German: ngerman
%   For English: english
%   For both German (main) and English: english,ngerman

\newcommand{\varIsDraft}{true}
% If the document is still in draft phase use 'true',
% if it is ready for submission use 'false'.
% The title page of a draft is marked as such, 
% a corresponding water mark is added on all other pages and
% overfull boxes (in margin space) are marked with black boxes.

\newcommand{\varIsConfidential}{false}
% Use 'true' if confidential content of the document shall be printed.

\newcommand{\varPaperSize}{A4}
% Possible formats include ISO formats A, B, C, and D, e.g. 'A4' and
% American formats, e.g. 'letter', 'legal', 'executive'

\newcommand{\varFontSize}{12pt}
% Font size of the main text, e.g. '10pt', '11pt', '12pt'

\newcommand{\varIsTwoSided}{true}
% Set whether document is intended for two-sided printing by using 'true',
% use 'false' for one-sided printing.

\newcommand{\varSpecifyPageLayoutManually}{false}
% Set how the document's page layout is determined.
% Use 'true' to specify the page layout manually using the geometry package
%   Use the \geometry{} command in customSettings.tex to set the desired layout
% Use 'false' if the layout should be constructed automatically (recommended)
%   It uses the values from
%       \varBindingWidth
%       \varDIV
%   The exact method of the automatic page layout construction method can be
%   found in scrguien.pdf: '2.2 Constructing the Type Area by Division'

\newcommand{\varBindingWidth}{5mm}
% Space occupied by book binding (e.g. glue binding, coil binding)
% Only taken into account for two-sided documents
% Not used for \varSpecifyPageLayoutManually = true

\newcommand{\varDIV}{calc}
% The DIV-factor specifies the number of strips into which the page is
% divided horizontally and vertically during the page layout construction
% Use a number '6', '7', ... '20', or 'calc'
% Not used for \varSpecifyPageLayoutManually = true

\newcommand{\varHasTwoColumns}{false}
% Set one-column typsetting by using 'false' or
% two-column typsetting by using 'true'.

\newcommand{\varParskip}{half}
% Set how the beginning of a new paragraph should be indicated.
% For inter-paragraph spacing use (not recommended)
%   'true' for one line vertical space or
%   'half' for for half line vertical space.
% For indentation in the first line without any vertical spacing use 'false' (recommended)
% Inter paragraph spacing has the drawback of disappearing, for instance, after a displayed formula.

\newcommand{\varBibFile}{SampleBib.bib}
